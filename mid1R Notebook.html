# Define two vectors
a <- c(10, 20, 30, 40)
b <- c(2, 4, 6, 8)

# Summation
sum_vec <- a + b
print(paste("Summation: ", toString(sum_vec)))
## [1] "Summation:  12, 24, 36, 48"
# Subtraction
sub_vec <- a - b
print(paste("Subtraction: ", sub_vec))
## [1] "Subtraction:  8"  "Subtraction:  16" "Subtraction:  24" "Subtraction:  32"
cat("sub",sub_vec)
## sub 8 16 24 32
# Multiplication
mul_vec <- a * b
print(paste("Multiplication: ", toString(mul_vec)))
## [1] "Multiplication:  20, 80, 180, 320"
# Division
div_vec <- a / b
print(paste("Division: ", toString(div_vec)))
## [1] "Division:  5, 5, 5, 5"

1. MatrixVector Multiplication

  • A matrix AA of size mn can be multiplied with a vector xx of size n1n1.

  • The result is a new vector yy of size m1m1.

  • Formula:

y=Axy=Ax

where

yi=j=1nAijxjyi=j=1nAijxj

A <- matrix(c(1,2,3,4,5,6), nrow=2, byrow=TRUE)
x <- c(1,2,3)
print(A)
##      [,1] [,2] [,3]
## [1,]    1    2    3
## [2,]    4    5    6
print(x)
## [1] 1 2 3
y <- A %*% x
print(y)
##      [,1]
## [1,]   14
## [2,]   32

MatrixVector Element-wise Multiplication

  • If vector length matches matrix dimension, element-wise multiplication occurs.

  • Formula:

Cij=AijxjCij=Aijxj

A <- matrix(c(1,2,3,4), nrow=2, byrow=TRUE)
x <- c(10,20)

result <- A * x   # vector recycled across rows
print(result)
##      [,1] [,2]
## [1,]   10   20
## [2,]   60   80
